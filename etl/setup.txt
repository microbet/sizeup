== Install a bootstrap virtualenv on ETL control client ==

# Do this so that all our python is separate from the Ubuntu system python. See https://stackoverflow.com/questions/4324558/whats-the-proper-way-to-install-pip-virtualenv-and-distribute-for-python

wget https://pypi.python.org/packages/d4/0c/9840c08189e030873387a73b90ada981885010dd9aea134d6de30cd24cb8/virtualenv-15.1.0.tar.gz#md5=44e19f4134906fe2d75124427dc9b716
tar xzf virtualenv-15.1.0.tar.gz
sudo mkdir -p /etc/sizeup/virtualenv
sudo python virtualenv-15.1.0/virtualenv.py /etc/sizeup/virtualenv/primary
sudo su

# as root:
apt-get install python-dev
. /etc/sizeup/virtualenv/primary/bin/activate
pip install download/virtualenv-15.1.0.tar.gz
pip install awscli --upgrade
pip install boto3
pip install ipython==5.4
